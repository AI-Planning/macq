188        """Converts the effects of an action as defined by tarski to fluents as defined by macq.
189
190        Args:

193

196        """

199        delete = set()
200        for effect in effects:
201            fluent = self.__tarski_atom_to_macq_fluent(effect.atom)

208    def __tarski_atom_to_macq_fluent(self, atom: Atom):
209        """Converts a tarski Atom to a fluent as defined by macq.
210
211        Args:
212            atom (Atom):
213                The supplied atom, defined using the tarski Atom class.
214
215        Returns:
216            A fluent, defined using the macq Fluent class.
217        """
218        # ignore functions for now
219        if not isinstance(atom, Atom):
220            return None
221        fluent_name = atom.predicate.name
222        terms = atom.subterms
223        objects = []
224        for term in terms:
225            if isinstance(fluent_name, BuiltinPredicateSymbol):
226                fluent_name = fluent_name.value
227            objects.append(PlanningObject(term.sort.name, term.name))
228        fluent = Fluent(fluent_name, objects)

253    def tarski_act_to_macq(self, tarski_act: PlainOperator):
254        """Converts an action as defined by tarski to an action as defined by macq.
255
256        Args:
257            tarski_act (PlainOperator):
258                The supplied action, defined using the tarski PlainOperator class.
259
260        Returns:
261            An action, defined using the macq Action class.
262        """
263        name_split = tarski_act.name.replace(")", "").split("(")
264        name = name_split[0]
265        obj_names = name_split[1].split(", ")
266
267        tarski_objs_mapping = {}
268        precond = set()
269        if isinstance(tarski_act.precondition, CompoundFormula):
270            raw_precond = tarski_act.precondition.subformulas
271            for raw_p in raw_precond:
272                if isinstance(raw_p, CompoundFormula):
273                    precond.add(self.__tarski_atom_to_macq_fluent(raw_p.subformulas[0]))
274                else:
275                    precond.add(self.__tarski_atom_to_macq_fluent(raw_p))
276        else:
277            precond.add(self.__tarski_atom_to_macq_fluent(tarski_act.precondition))
278        (add, delete) = self.__effect_split(tarski_act)
279
280        tarski_objs_mapping.update({o.name:o for fluent in add for o in fluent.objects })
281        tarski_objs_mapping.update({o.name:o for fluent in delete for o in fluent.objects })
282        tarski_objs_mapping.update({o.name:o for fluent in precond for o in fluent.objects })
283
284        obj_params = [tarski_objs_mapping[o] for o in obj_names]
285
286        return (
287            Action(
288                name=name,
289                obj_params=obj_params,
290                precond=precond,
291                add=add,
292                delete=delete,
293            )
294            if self.observe_pres_effs
295            else Action(name=name, obj_params=obj_params)
296        )

442    def generate_single_trace_from_plan(self, plan: Plan):
443        """Generates a single trace from the plan taken as input.
444
445        Args:
446            plan (Plan):
447                The plan to generate a trace from.
448
449        Returns:
450            The trace generated from the plan.
451        """
452        trace = Trace()
453        actions = plan.actions
454        plan_len = len(actions)
455        # get initial state
456        state = self.problem.init
457        # note that we add 1 because the states represented take place BEFORE their subsequent action,
458        # so if we need to take x actions, we need x + 1 states and therefore x + 1 steps in the trace.
459        for i in range(plan_len + 1):
460            macq_state = self.tarski_state_to_macq(state)
461            # if we have not yet reached the end of the trace
462            if len(trace) < plan_len:
463                act = actions[i]
464                trace.append(Step(macq_state, self.tarski_act_to_macq(act), i + 1))
465                state = progress(state, act)
466            else:
467                trace.append(Step(macq_state, None, i + 1))
468        return trace
